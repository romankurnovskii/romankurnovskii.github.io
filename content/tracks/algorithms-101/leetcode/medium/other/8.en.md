```js
// .js
/**
 * @param {string} s
 * @return {number}
 */
var myAtoi = function(s) {
    let res = 0;
    let num = parseInt(s);
    if(num >= 2147483648){
        res = 2147483647;
    } else if (num <= -2147483648){
        res = -2147483648;
    } else if (isNaN(num)) {
        res = 0;
    } else {
        res = num;
    }

    return res;
};
```
