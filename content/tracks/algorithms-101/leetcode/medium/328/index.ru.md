```python
class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next

def oddEvenList(head: ListNode) -> ListNode:
    if not head:
        return head
    
    odd = head
    even = head.next
    even_head = even
    
    while even and even.next:
        odd.next = even.next
        odd = odd.next
        even.next = odd.next
        even = even.next
    
    # –°–æ–µ–¥–∏–Ω—è–µ–º –ø–æ—Å–ª–µ–¥–Ω–∏–π –Ω–µ—á–µ—Ç–Ω—ã–π —É–∑–µ–ª —Å –ø–µ—Ä–≤—ã–º —á–µ—Ç–Ω—ã–º
    odd.next = even_head
    
    return head
```
