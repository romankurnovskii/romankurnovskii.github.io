## –†–µ—à–µ–Ω–∏–µ

class Solution:
    def oddEvenList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        odd_head = ListNode(0)
        even_head = ListNode(0)

        odd = odd_head
        even = even_head
        is_odd = True
        while head:
            if is_odd:
                odd.next = head
                odd = odd.next
            else:
                even.next = head
                even = even.next

            head = head.next    
            is_odd = not is_odd
        
        even.next = None  # —Å–∞–º—ã–π –ø–æ—Å–ª–µ–¥–Ω–∏–π —É–∑–µ–ª
        odd.next = even_head.next # head –≤ –∫–æ–Ω–µ—Ü —Å–ø–∏—Å–∫–∞

        return odd_head.next
```

class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next

def oddEvenList(head: ListNode) -> ListNode:
    if not head:
        return head
    
    odd = head
    even = head.next

    even_head = even
    
    while even and even.next:
        odd.next = even.next
        odd = odd.next
        even.next = odd.next
        even = even.next
    
    # –°–æ–µ–¥–∏–Ω—è–µ–º –ø–æ—Å–ª–µ–¥–Ω–∏–π –Ω–µ—á–µ—Ç–Ω—ã–π —É–∑–µ–ª —Å –ø–µ—Ä–≤—ã–º —á–µ—Ç–Ω—ã–º
    odd.next = even_head
    
    return head
```
