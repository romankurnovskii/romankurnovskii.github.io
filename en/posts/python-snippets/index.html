 504# Python offers a fundamental abstraction called the Iterable.
 505# An iterable is an object that can be treated as a sequence.
 506# The object returned by the range function, is an iterable.
 507
 508filled_dict = {"one": 1, "two": 2, "three": 3}
 509our_iterable = filled_dict.keys()
 510print(our_iterable)  # => dict_keys(['one', 'two', 'three']). This is an object that implements our Iterable interface.
 511
 512# We can loop over it.
 513for i in our_iterable:
 514    print(i)  # Prints one, two, three
 515
 516# However we cannot address elements by index.
 517our_iterable[1]  # Raises a TypeError
 518
 519# An iterable is an object that knows how to create an iterator.
 520our_iterator = iter(our_iterable)
 521
 522# Our iterator is an object that can remember the state as we traverse through it.
 523# We get the next object with "next()".
 524next(our_iterator)  # => "one"
 525
 526# It maintains state as we iterate.
 527next(our_iterator)  # => "two"
 528next(our_iterator)  # => "three"
 529
 530# After the iterator has returned all of its data, it raises a StopIteration exception
 531next(our_iterator)  # Raises StopIteration
 532
 533# We can also loop over it, in fact, "for" does this implicitly!
 534our_iterator = iter(our_iterable)
 535for i in our_iterator:
 536    print(i)  # Prints one, two, three
 537
 538# You can grab all the elements of an iterable or iterator by calling list() on it.
 539list(our_iterable)  # => Returns ["one", "two", "three"]
 540list(our_iterator)  # => Returns [] because state is saved

 621def create_adder(x):
 622    def adder(y):
 623        return x + y
 624    return adder

 655print(math.sqrt(16))  # => 4.0
 656
 657# You can get specific functions from a module
 658from math import ceil, floor
 659print(ceil(3.7))   # => 4.0
 660print(floor(3.7))  # => 3.0

 663# Warning: this is not recommended
 664from math import *
 665

 667import math as m
 668math.sqrt(16) == m.sqrt(16)  # => True

 671# can write your own, and import them. The name of the
 672# module is the same as the name of the file.
 673
 674# You can find out which functions and attributes
 675# are defined in a module.
 676import math
 677dir(math)

 680# folder as your current script, the file math.py will
 681# be loaded instead of the built-in Python module.

 687## 6. Classes
 688####################################################
 689
 690# We use the "class" statement to create a class
 691class Human:
 692
 693    # A class attribute. It is shared by all instances of this class
 694    species = "H. sapiens"
 695
 696    # Basic initializer, this is called when this class is instantiated.
 697    # Note that the double leading and trailing underscores denote objects
 698    # or attributes that are used by Python but that live in user-controlled
 699    # namespaces. Methods(or objects or attributes) like: __init__, __str__,
 700    # __repr__ etc. are called special methods (or sometimes called dunder methods)
 701    # You should not invent such names on your own.
 702    def __init__(self, name):
 703        # Assign the argument to the instance's name attribute
 704        self.name = name
 705
 706        # Initialize property
 707        self._age = 0
 708
 709    # An instance method. All methods take "self" as the first argument
 710    def say(self, msg):
 711        print("{name}: {message}".format(name=self.name, message=msg))
 712
 713    # Another instance method
 714    def sing(self):
 715        return 'yo... yo... microphone check... one two... one two...'
 716
 717    # A class method is shared among all instances
 718    # They are called with the calling class as the first argument
 719    @classmethod
 720    def get_species(cls):
 721        return cls.species
 722
 723    # A static method is called without a class or instance reference
 724    @staticmethod
 725    def grunt():
 726        return "*grunt*"
 727
 728    # A property is just like a getter.
 729    # It turns the method age() into a read-only attribute of the same name.
 730    # There's no need to write trivial getters and setters in Python, though.
 731    @property
 732    def age(self):
 733        return self._age
 734
 735    # This allows the property to be set
 736    @age.setter
 737    def age(self, age):
 738        self._age = age
 739
 740    # This allows the property to be deleted
 741    @age.deleter
 742    def age(self):
 743        del self._age
 744
 745
 746# When a Python interpreter reads a source file it executes all its code.
 747# This __name__ check makes sure this code block is only executed when this
 748# module is the main program.
 749if __name__ == '__main__':
 750    # Instantiate a class
 751    i = Human(name="Ian")
 752    i.say("hi")                     # "Ian: hi"
 753    j = Human("Joel")
 754    j.say("hello")                  # "Joel: hello"
 755    # i and j are instances of type Human, or in other words: they are Human objects
 756
 757    # Call our class method
 758    i.say(i.get_species())          # "Ian: H. sapiens"
 759    # Change the shared attribute
 760    Human.species = "H. neanderthalensis"
 761    i.say(i.get_species())          # => "Ian: H. neanderthalensis"
 762    j.say(j.get_species())          # => "Joel: H. neanderthalensis"
 763
 764    # Call the static method
 765    print(Human.grunt())            # => "*grunt*"
 766
 767    # Static methods can be called by instances too
 768    print(i.grunt())                # => "*grunt*"
 769
 770    # Update the property for this instance
 771    i.age = 42
 772    # Get the property
 773    i.say(i.age)                    # => "Ian: 42"
 774    j.say(j.age)                    # => "Joel: 0"
 775    # Delete the property
 776    del i.age
 777    # i.age                         # => this would raise an AttributeError

 781## 6.1 Inheritance
 782####################################################
 783
 784# Inheritance allows new child classes to be defined that inherit methods and
 785# variables from their parent class.
 786
 787# Using the Human class defined above as the base or parent class, we can
 788# define a child class, Superhero, which inherits the class variables like
 789# "species", "name", and "age", as well as methods, like "sing" and "grunt"
 790# from the Human class, but can also have its own unique properties.
 791
 792# To take advantage of modularization by file you could place the classes above in their own files,
 793# say, human.py
 794
 795# To import functions from other files use the following format
 796# from "filename-without-extension" import "function-or-class"
 797
 798from human import Human
 799
 800
 801# Specify the parent class(es) as parameters to the class definition
 802class Superhero(Human):
 803
 804    # If the child class should inherit all of the parent's definitions without
 805    # any modifications, you can just use the "pass" keyword (and nothing else)
 806    # but in this case it is commented out to allow for a unique child class:
 807    # pass
 808
 809    # Child classes can override their parents' attributes
 810    species = 'Superhuman'
 811
 812    # Children automatically inherit their parent class's constructor including
 813    # its arguments, but can also define additional arguments or definitions
 814    # and override its methods such as the class constructor.
 815    # This constructor inherits the "name" argument from the "Human" class and
 816    # adds the "superpower" and "movie" arguments:
 817    def __init__(self, name, movie=False,
 818                 superpowers=["super strength", "bulletproofing"]):
 819
 820        # add additional class attributes:
 821        self.fictional = True
 822        self.movie = movie
 823        # be aware of mutable default values, since defaults are shared
 824        self.superpowers = superpowers
 825
 826        # The "super" function lets you access the parent class's methods
 827        # that are overridden by the child, in this case, the __init__ method.
 828        # This calls the parent class constructor:
 829        super().__init__(name)
 830
 831    # override the sing method
 832    def sing(self):
 833        return 'Dun, dun, DUN!'
 834
 835    # add an additional instance method
 836    def boast(self):
 837        for power in self.superpowers:
 838            print("I wield the power of {pow}!".format(pow=power))
 839
 840
 841if __name__ == '__main__':
 842    sup = Superhero(name="Tick")
 843
 844    # Instance type checks
 845    if isinstance(sup, Human):
 846        print('I am human')
 847    if type(sup) is Superhero:
 848        print('I am a superhero')
 849
 850    # Get the Method Resolution search Order used by both getattr() and super()
 851    # This attribute is dynamic and can be updated
 852    print(Superhero.__mro__)    # => (<class '__main__.Superhero'>,
 853                                # => <class 'human.Human'>, <class 'object'>)
 854
 855    # Calls parent method but uses its own class attribute
 856    print(sup.get_species())    # => Superhuman
 857
 858    # Calls overridden method
 859    print(sup.sing())           # => Dun, dun, DUN!
 860
 861    # Calls method from Human
 862    sup.say('Spoon')            # => Tick: Spoon
 863
 864    # Call method that exists only in Superhero
 865    sup.boast()                 # => I wield the power of super strength!
 866                                # => I wield the power of bulletproofing!
 867
 868    # Inherited class attribute
 869    sup.age = 31
 870    print(sup.age)              # => 31
 871
 872    # Attribute that only exists within Superhero
 873    print('Am I Oscar eligible? ' + str(sup.movie))

 881class Bat:
 882
 883    species = 'Baty'

 885    def __init__(self, can_fly=True):
 886        self.fly = can_fly
 887
 888    # This class also has a say method
 889    def say(self, msg):

 891        return msg

 898    b = Bat()

 900    print(b.fly)

 902
 903# And yet another class definition that inherits from Superhero and Bat
 904# superhero.py
 905from superhero import Superhero
 906from bat import Bat
 907
 908# Define Batman as a child that inherits from both Superhero and Bat
 909class Batman(Superhero, Bat):
 910
 911    def __init__(self, *args, **kwargs):
 912        # Typically to inherit attributes you have to call super:
 913        # super(Batman, self).__init__(*args, **kwargs)
 914        # However we are dealing with multiple inheritance here, and super()
 915        # only works with the next base class in the MRO list.
 916        # So instead we explicitly call __init__ for all ancestors.
 917        # The use of *args and **kwargs allows for a clean way to pass arguments,
 918        # with each parent "peeling a layer of the onion".
 919        Superhero.__init__(self, 'anonymous', movie=True,
 920                           superpowers=['Wealthy'], *args, **kwargs)
 921        Bat.__init__(self, *args, can_fly=False, **kwargs)
 922        # override the value for the name attribute
 923        self.name = 'Sad Affleck'
 924
 925    def sing(self):
 926        return 'nan nan nan nan nan batman!'
 927
 928
 929if __name__ == '__main__':
 930    sup = Batman()
 931
 932    # Get the Method Resolution search Order used by both getattr() and super().
 933    # This attribute is dynamic and can be updated
 934    print(Batman.__mro__)       # => (<class '__main__.Batman'>,
 935                                # => <class 'superhero.Superhero'>,
 936                                # => <class 'human.Human'>,
 937                                # => <class 'bat.Bat'>, <class 'object'>)
 938
 939    # Calls parent method but uses its own class attribute
 940    print(sup.get_species())    # => Superhuman
 941
 942    # Calls overridden method
 943    print(sup.sing())           # => nan nan nan nan nan batman!
 944
 945    # Calls method from Human, because inheritance order matters
 946    sup.say('I agree')          # => Sad Affleck: I agree
 947
 948    # Call method that exists only in 2nd ancestor
 949    print(sup.sonar())          # => ))) ... (((
 950
 951    # Inherited class attribute
 952    sup.age = 100
 953    print(sup.age)              # => 100
 954
 955    # Inherited attribute from 2nd ancestor whose default value was overridden.
 956    print('Can I fly? ' + str(sup.fly)) # => Can I fly? False

 964# Generators help you make lazy code.
 965def double_numbers(iterable):
 966    for i in iterable:
 967        yield i + i
 968
 969# Generators are memory-efficient because they only load the data needed to
 970# process the next value in the iterable. This allows them to perform
 971# operations on otherwise prohibitively large value ranges.
 972# NOTE: `range` replaces `xrange` in Python 3.
 973for i in double_numbers(range(1, 900000000)):  # `range` is a generator.
 974    print(i)
 975    if i >= 30:
 976        break
 977
 978# Just as you can create a list comprehension, you can create generator
 979# comprehensions as well.
 980values = (-x for x in [1,2,3,4,5])
 981for x in values:
 982    print(x)  # prints -1 -2 -3 -4 -5 to console/terminal
 983
 984# You can also cast a generator comprehension directly to a list.
 985values = (-x for x in [1,2,3,4,5])
 986gen_to_list = list(values)
 987print(gen_to_list)  # => [-1, -2, -3, -4, -5]

 990# Decorators
 991# In this example `beg` wraps `say`. If say_please is True then it
 992# will change the returned message.
 993from functools import wraps
 994
 995
 996def beg(target_function):
 997    @wraps(target_function)
 998    def wrapper(*args, **kwargs):
 999        msg, say_please = target_function(*args, **kwargs)
1000        if say_please:
1001            return "{} {}".format(msg, "Please! I am poor :(")
1002        return msg
1003
1004    return wrapper
1005
1006
1007@beg
1008def say(say_please=False):
1009    msg = "Can you buy me a beer?"
1010    return msg, say_please
1011
1012
1013print(say())                 # Can you buy me a beer?
1014print(say(say_please=True))  # Can you buy me a beer? Please! I am poor :(
