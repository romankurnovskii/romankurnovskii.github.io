546    return this.charAt(0);

550// This fact is often used in "polyfilling", which is implementing newer
551// features of JavaScript in an older subset of JavaScript, so that they can be
552// used in older environments such as outdated browsers.
553
554// For instance, we mentioned that Object.create isn't yet available in all
555// implementations, but we can still use it with this polyfill:
556if (Object.create === undefined){ // don't overwrite it if it exists
557    Object.create = function(proto){
558        // make a temporary constructor with the right prototype
559        var Constructor = function(){};
560        Constructor.prototype = proto;
561        // then use it to create a new, appropriately-prototyped object
562        return new Constructor();
563    };
564}
