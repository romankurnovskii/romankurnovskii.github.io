Fenwick Tree

 1class Fenwick: #also known as Binary Indexed Tree (BIT)
 2    def __init__(self, n):
 3        self.n = n
 4        self.bit = [0] * (n+1)
 5
 6    def add(self, idx, val):
 7        while idx <= self.n:
 8            self.bit[idx] += val
 9            idx += idx & -idx
10
11    def add_range(self, l, r, val):
12        self.add(l, val)
13        self.add(r+1, -val)
14
15    def query(self, idx):
16        #Calculates the sum of the elements from the beginning to idx
17        ret = 0
18        while idx > 0:
19            ret += self.bit[idx]
20            idx -= idx & -idx
21        return ret
22
23    def range_sum(self, l, r):
24        # Return the sum of the elements from l (inclusive) to r (exclusive)
25        return self.prefix_sum(r - 1) - self.prefix_sum(l - 1)
26
27    def prefix_sum(self, x):
28        # return sum upto and including element x
29        z = x
30        res = 0
31        while z >= 0:
32            res += self.bit[z]
33            # Strip trailing zeros from z, and then take away one
34            z = (z & (z + 1)) - 1
35        return res
