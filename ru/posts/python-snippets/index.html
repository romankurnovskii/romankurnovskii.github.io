 689    # Обычный конструктор, вызывается при инициализации экземпляра класса
 690    # Обратите внимание, что двойное подчёркивание в начале и в конце имени
 691    # означает объекты и атрибуты, которые используются Python, но находятся
 692    # в пространствах имён, управляемых пользователем.
 693    # Методы (или объекты или атрибуты), например:
 694    # __init__, __str__, __repr__ и т. д. называются специальными методами.

 697        # Присваивание значения аргумента атрибуту

 700        # Инициализация свойства

 782# Используя класс Human, определенный выше как базовый или родительский класс, мы можем
 783# определить дочерний класс Superhero, который наследует переменные класса, такие как
 784# "species", "name" и "age", а также методы, такие как "sing" и "grunt" из класса Human,
 785# но также может иметь свои уникальные свойства.
 786
 787# Чтобы воспользоваться преимуществами модульности по файлам, вы можете поместить
 788# вышеперечисленные классы в их собственные файлы, например, human.py

 790# Чтобы импортировать функции из других файлов, используйте следующий формат

 796# Укажите родительский класс(ы) как параметры определения класса
 797class Superhero(Human):
 798
 799    # Если дочерний класс должен наследовать все определения родителя без каких-либо
 800    # изменений, вы можете просто использовать ключевое слово pass (и ничего больше),
 801    # но в этом случае оно закомментировано, чтобы разрешить уникальный дочерний класс:
 802    # pass
 803
 804    # Дочерние классы могут переопределять атрибуты своих родителей
 805    species = 'Сверхчеловек'
 806
 807    # Дочерние классы автоматически наследуют конструктор родительского класса, включая
 808    # его аргументы, но также могут определять дополнительные аргументы или определения
 809    # и переопределять его методы, такие как конструктор класса.
 810    # Этот конструктор наследует аргумент "name" от класса "Human"
 811    # и добавляет аргументы "superpower" и "movie":
 812    def __init__(self, name, movie=False,
 813                 superpowers=["сверхсила", "пуленепробиваемость"]):
 814
 815        # добавить дополнительные атрибуты класса:
 816        self.fictional = True
 817        self.movie = movie
 818        # помните об изменяемых значениях по умолчанию,
 819        # поскольку значения по умолчанию являются общими
 820        self.superpowers = superpowers
 821
 822        # Функция "super" позволяет вам получить доступ к методам родительского класса,
 823        # которые переопределяются дочерним, в данном случае, методом __init__.
 824        # Это вызывает конструктор родительского класса:
 825        super().__init__(name)
 826
 827    # переопределить метод sing
 828    def sing(self):
 829        return 'Бам, бам, БАМ!'
 830
 831    # добавить дополнительный метод экземпляра
 832    def boast(self):
 833        for power in self.superpowers:
 834            print("Я обладаю силой '{pow}'!".format(pow=power))

 848    # Этот атрибут является динамическим и может быть обновлен

 863                                # => Я обладаю силой 'пуленепробиваемость'!

 901# И еще одно определение класса, унаследованное от Superhero и Bat
 902# superhero.py
 903from superhero import Superhero
 904from bat import Bat
 905
 906# Определите Batman как дочерний класс, унаследованный от Superhero и Bat
 907class Batman(Superhero, Bat):
 908
 909    def __init__(self, *args, **kwargs):
 910        # Обычно для наследования атрибутов необходимо вызывать super:
 911        # super(Batman, self).__init__(*args, **kwargs)
 912        # Однако здесь мы имеем дело с множественным наследованием, а super()
 913        # работает только со следующим базовым классом в списке MRO.
 914        # Поэтому вместо этого мы вызываем __init__ для всех родителей.
 915        # Использование *args и **kwargs обеспечивает чистый способ передачи
 916        # аргументов, когда каждый родитель "очищает слой луковицы".
 917        Superhero.__init__(self, 'анонимный', movie=True,
 918                           superpowers=['Богатый'], *args, **kwargs)
 919        Bat.__init__(self, *args, can_fly=False, **kwargs)
 920        # переопределить значение атрибута name
 921        self.name = 'Грустный Бен Аффлек'
 922
 923    def sing(self):
 924        return 'на на на на на бэтмен!'
 925
 926
 927if __name__ == '__main__':
 928    sup = Batman()
 929
 930    # Получить порядок поиска разрешения метода (MRO),
 931    # используемый как getattr(), так и super()
 932    # Этот атрибут является динамическим и может быть обновлен
 933    print(Batman.__mro__)       # => (<class '__main__.Batman'>,
 934                                # => <class 'superhero.Superhero'>,
 935                                # => <class 'human.Human'>,
 936                                # => <class 'bat.Bat'>, <class 'object'>)
 937
 938    # Вызывает родительский метод, но использует свой собственный атрибут класса
 939    print(sup.get_species())    # => Сверхчеловек
 940
 941    # Вызов переопределенного метода
 942    print(sup.sing())           # => на на на на на бэтмен!
 943
 944    # Вызывает метод из Human, потому что порядок наследования имеет значение
 945    sup.say('Я согласен')          # => Грустный Бен Аффлек: Я согласен
 946
 947    # Вызов метода, существующий только во втором родителе
 948    print(sup.sonar())          # => ))) ... (((
 949
 950    # Атрибут унаследованного класса
 951    sup.age = 100
 952    print(sup.age)              # => 100
 953
 954    # Унаследованный атрибут от второго родителя,
 955    # значение по умолчанию которого было переопределено.
 956    print('Могу ли я летать? ' + str(sup.fly)) # => Могу ли я летать? False

 962
 963# Генераторы помогут выполнить ленивые вычисления
 964def double_numbers(iterable):
 965    for i in iterable:
 966        yield i + i
 967
 968# Генераторы эффективны с точки зрения памяти, потому что они загружают только данные,
 969# необходимые для обработки следующего значения в итерации.
 970# Это позволяет им выполнять операции с недопустимо большими диапазонами значений.
 971# ПРИМЕЧАНИЕ: "range" заменяет "xrange" в Python 3.
 972for i in double_numbers(range(1, 900000000)):  # "range" - генератор.
 973    print(i)
 974    if i >= 30:
 975        break
 976
 977# Так же, как вы можете создать интерпретации списков, вы можете создать и
 978# интерпретации генераторов.
 979values = (-x for x in [1,2,3,4,5])
 980for x in values:
 981    print(x)  # Выводит -1 -2 -3 -4 -5
 982
 983# Вы также можете преобразовать интерпретацию генератора непосредственно в список.
 984values = (-x for x in [1,2,3,4,5])
 985gen_to_list = list(values)
 986print(gen_to_list)  # => [-1, -2, -3, -4, -5]

 989# Декораторы
 990# В этом примере "beg" оборачивает "say".
 991# Если say_please равно True, он изменит возвращаемое сообщение.
 992from functools import wraps
 993
 994
 995def beg(target_function):
 996    @wraps(target_function)
 997    def wrapper(*args, **kwargs):
 998        msg, say_please = target_function(*args, **kwargs)
 999        if say_please:
1000            return "{} {}".format(msg, "Пожалуйста! Спасибо :)")
1001        return msg
1002
1003    return wrapper
1004
1005
1006@beg
1007def say(say_please=False):
1008    msg = "Вы не купите мне сока?"
1009    return msg, say_please
1010
1011
1012print(say())                 # Вы не купите мне сока?
1013print(say(say_please=True))  # Вы не купите мне сока? Пожалуйста! Спасибо :)
